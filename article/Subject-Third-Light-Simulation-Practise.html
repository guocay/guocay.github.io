	private static class LightStatus {

			// å¦‚æœæ“ä½œåçš„ç»“æœå°äº -1, åˆ™å›é€€è‡³ -1.
			// å¦‚æœæ“ä½œåçš„ç»“æœå¤§äº 4, åˆ™å›é€€è‡³ 4.
			this.lever = max(min(this.lever + lever, CONFIG_MAX_ALTERNATE_COUNT), LEVER_PARALLEL_LIGHT_STATUS);
		}

		public void setEmergency(int emergency) {
			// åˆ¤æ–­æ“ä½œçš„æ¬¡æ•°æ˜¯å¥‡æ•°è¿˜æ˜¯å¶æ•°, æ˜¯å¥‡æ•°åˆ™ä¸æ”¹å˜ç¯å…‰çŠ¶æ€. åä¹‹åˆ™æ”¹å˜çŠ¶æ€.

		}
